  })

let lobbyRoomId = 42

var lobby: PCRoom

currentUser.joinRoom(roomID: lobbyRoomId) { error in

    lobby = currentUser.rooms[0]

    // Hello, everyone! 👋

}
lateinit var lobby: Room

currentUser.joinRoom(

  roomId = "42",

  callback = { result ->

    when (result) {

      is Result.Success -> {

        lobby = result.value

        // Hello, everyone! 👋

      }

    }

  }

)
